//저장소 생성 및 연결
$git init 
	저장소 생성하기 
	프로젝트 폴더에 숨김파일로 .git 생성
$git remote add origin [원격저장소 주소] 
	[원격저장소 주소] 깃허브에서 new repository 생성하고 해당주소 입력
	깃허브와 .git이 연결
$git branch -m master main 
	master에 main branch 생성

//파일 업로드
$git pull (또는 git pull origin [브랜치이름])
	업로드할 브랜치 불러오기 입력안할 시 현재 브랜치	
$git add .
	commit할 파일들 담기	
$git commit -m "massage" 	
	commit하고 메세지 적기	
$git push (또는 git push origin [브랜치이름])
	최종적으로 github에 올리기
	
//추가적인 명령어
$git remote -v
$git remote rm origin
$git branch
$git config --global init.defaultBranch [브랜치이름]
$git status
	현재 상태 확인
$git rm --cached -r .
